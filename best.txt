function loadGames(list){
const container=document.getElementById('games');
container.innerHTML='';
list.forEach(g=>{
const card=document.createElement('div');
card.className='game';
card.innerHTML=`<img src="${g.img}"><h3>${g.name}</h3>`;
card.onclick=()=>playGame(g.url);
container.appendChild(card);
});
}
function playGame(url){
 document.getElementById('player').style.display='flex';
 document.getElementById('frame').src=url;
}
function closeGame(){
 document.getElementById('player').style.display='none';
 document.getElementById('frame').src='';
}
function searchGames(){
const text=document.getElementById('search').value.toLowerCase();
const filtered=games.filter(g=>g.name.toLowerCase().includes(text));
loadGames(filtered);
}
function filter(cat){
if(cat==='all'){loadGames(games);return}
loadGames(games.filter(g=>g.cat===cat));
}
loadGames(games);
